# List unpacking
a, b, c = [1, 2, 3]
print(a, b, c) # output: 1 2 3

# Tuple unpacking
x, y, z = (4, 5, 6)
print(x, y, z) # output: 4 5 6

# String unpacking
s1, s2 = "ab"
print(s1, s2) # output: a b

# Dict unpacking
k1, k2 = {"a": 1, "b": 2}
print(k1, k2) # output: a b
